function toggleWarehouseEdit() {
    const warehouseCodeInput = document.getElementById('warehouse_code');
    const editButton = event.target.closest('button');
    
    if (warehouseCodeInput.readOnly) {
        warehouseCodeInput.readOnly = false;
        warehouseCodeInput.focus();
        editButton.innerHTML = ' Save';
        editButton.classList.remove('btn-outline-secondary');
        editButton.classList.add('btn-outline-success');
    } else {
        warehouseCodeInput.readOnly = true;
        editButton.innerHTML = ' Edit';
        editButton.classList.remove('btn-outline-success');
        editButton.classList.add('btn-outline-secondary');
        
        // Auto-populate bin location based on warehouse
        const binLocationInput = document.getElementById('bin_location');
        if (warehouseCodeInput.value) {
            binLocationInput.value = warehouseCodeInput.value + '-BIN-01'; // Default bin
        }
    }
    
    // Refresh feather icons
    setTimeout(() => {
        if (window.feather) {
            feather.replace();
        }
    }, 100);
}

// Show JSON Preview for SAP B1 posting
function showJsonPreview(grpoId) {
    console.log('🔍 Fetching JSON preview for GRPO ID:', grpoId);
    
    fetch(`/api/grpo/${grpoId}/preview_json`)
        .then(response => {
            console.log('📡 Response status:', response.status);
            if (!response.ok) {
                throw new Error(`HTTP ${response.status}: ${response.statusText}`);
            }
            return response.json();
        })
        .then(data => {
            console.log('📄 JSON Preview Response:', data);
            
            const jsonContent = document.getElementById('jsonContent');
            if (data.success) {
                const formattedJson = JSON.stringify(data.json_data, null, 2);
                jsonContent.textContent = formattedJson;
                
                console.log('✅ JSON Structure for SAP B1:');
                console.log('📊 PO Number:', data.po_number);
                console.log('📋 Total Lines:', data.total_lines);
                console.log('🏗️ Complete JSON Structure:');
                console.log(formattedJson);
                
                // Also log individual components for debugging
                if (data.json_data) {
                    console.log('🏢 Card Code:', data.json_data.CardCode);
                    console.log('📅 Doc Date:', data.json_data.DocDate);
                    console.log('🏪 Business Place ID:', data.json_data.BPL_IDAssignedToInvoice);
                    console.log('📦 Document Lines Count:', data.json_data.DocumentLines?.length || 0);
                    
                    if (data.json_data.DocumentLines) {
                        data.json_data.DocumentLines.forEach((line, index) => {
                            console.log(`📝 Line ${index + 1}:`, {
                                ItemCode: line.ItemCode,
                                Quantity: line.Quantity,
                                WarehouseCode: line.WarehouseCode,
                                BaseType: line.BaseType,
                                BaseEntry: line.BaseEntry,
                                BaseLine: line.BaseLine
                            });
                        });
                    }
                }
            } else {
                const errorMsg = `❌ Error generating JSON preview: ${data.error}`;
                jsonContent.textContent = errorMsg;
                console.error('❌ JSON Preview Error:', data.error);
            }
            
            const modal = new bootstrap.Modal(document.getElementById('jsonPreviewModal'));
            modal.show();
        })
        .catch(error => {
            console.error('🚨 Error fetching JSON preview:', error);
            const jsonContent = document.getElementById('jsonContent');
            const errorMsg = `🚨 Network/Server Error: ${error.message}`;
            jsonContent.textContent = errorMsg;
            
            const modal = new bootstrap.Modal(document.getElementById('jsonPreviewModal'));
            modal.show();
        });
}

function copyJsonToClipboard() {
    const jsonContent = document.getElementById('jsonContent').textContent;
    navigator.clipboard.writeText(jsonContent).then(() => {
        alert('JSON copied to clipboard!');
    }).catch(err => {
        console.error('Could not copy text: ', err);
        alert('Failed to copy JSON to clipboard');
    });
}

function submitGRPO(grpoId) {
    if (confirm('Are you sure you want to submit this GRPO for QC approval?')) {
        fetch(`/grpo/${grpoId}/submit`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            }
        })
        .then(response => {
            if (response.ok) {
                location.reload();
            } else {
                alert('Error submitting GRPO');
            }
        })
        .catch(error => {
            console.error('Error:', error);
            alert('Error submitting GRPO');
        });
    }
}

function approveGRPO(grpoId) {
    const notes = prompt('QC Notes (optional):');
    if (notes !== null) {
        const draftOrPost = confirm('Click OK to POST to SAP B1 immediately, or Cancel to save as DRAFT for later posting');
        
        const formData = new FormData();
        formData.append('qc_notes', notes);
        formData.append('draft_or_post', draftOrPost ? 'post' : 'draft');
        
        fetch(`/grpo/${grpoId}/approve`, {
            method: 'POST',
            body: formData
        })
        .then(response => {
            if (response.ok) {
                location.reload();
            } else {
                alert('Error approving GRPO');
            }
        })
        .catch(error => {
            console.error('Error:', error);
            alert('Error approving GRPO');
        });
    }
}

function rejectGRPO(grpoId) {
    const notes = prompt('Rejection reason (required):');
    if (notes && notes.trim()) {
        const formData = new FormData();
        formData.append('qc_notes', notes);
        
        fetch(`/grpo/${grpoId}/reject`, {
            method: 'POST',
            body: formData
        })
        .then(response => {
            if (response.ok) {
                location.reload();
            } else {
                alert('Error rejecting GRPO');
            }
        })
        .catch(error => {
            console.error('Error:', error);
            alert('Error rejecting GRPO');
        });
    } else {
        alert('Rejection reason is required');
    }
}

function printBarcode(barcode, itemId) {
    // Call the print barcode API

        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({
            barcode: barcode,
            item_id: itemId
        })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            alert(data.message);
            // Reload page to update UI
            location.reload();
        } else {
            alert('Error printing barcode: ' + data.error);
        }
    })
    .catch(error => {
        console.error('Error:', error);
        alert('Error printing barcode');
    });
}

// Generate barcode for the modal form
function generateBarcodeForModal() {
    const itemCodeInput = document.getElementById('item_code');
    const barcodeInput = document.getElementById('barcode');
    
    if (!itemCodeInput.value.trim()) {
        alert('Please enter an item code first');
        itemCodeInput.focus();
        return;
    }
    
    // Generate WMS barcode format
    const itemCode = itemCodeInput.value.trim();
    const randomHex = Math.random().toString(16).substring(2, 10).toUpperCase();
    const generatedBarcode = `WMS-${itemCode}-${randomHex}`;
    
    barcodeInput.value = generatedBarcode;
    console.log('🏷️ Generated barcode:', generatedBarcode);
}

// Generate barcode for existing GRPO items
function generateBarcode(itemId, itemCode) {
    console.log('🏷️ Generating barcode for item:', itemCode, 'ID:', itemId);
    
    // Generate WMS barcode format
    const randomHex = Math.random().toString(16).substring(2, 10).toUpperCase();
    const generatedBarcode = `WMS-${itemCode}-${randomHex}`;
    
    // Update the item with the generated barcode
    updateItemField(itemId, 'generated_barcode', generatedBarcode);
}

// Update item field via AJAX
function updateItemField(itemId, fieldName, value) {
    console.log('📝 Updating item field:', itemId, fieldName, value);
    
    fetch(`/grpo/item/${itemId}/update_field`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({
            field_name: fieldName,
            value: value
        })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            console.log('✅ Field updated successfully');
            // Refresh the page to show updates
            location.reload();
        } else {
            console.error('❌ Error updating field:', data.error);
            alert('Error updating field: ' + data.error);
        }
    })
    .catch(error => {
        console.error('❌ Network error:', error);
        alert('Network error updating field');
    });
}

// Add event listeners when page loads
document.addEventListener('DOMContentLoaded', function() {
    console.log('🚀 Page loaded, setting up event listeners...');
    
    // Add event listeners for Add Item buttons
    document.querySelectorAll('.add-item-btn').forEach(button => {
        button.addEventListener('click', function() {
            console.log('🔘 Add button clicked');
            
            const itemCode = this.getAttribute('data-item-code');
            const itemName = this.getAttribute('data-item-name');
            const uom = this.getAttribute('data-uom');
            const warehouse = this.getAttribute('data-warehouse');
            const poItemJson = this.getAttribute('data-po-item');
            
            console.log('📋 Button data:', {
                itemCode, itemName, uom, warehouse, poItemJson
            });
            
            // Validate required data
            if (!itemCode) {
                console.error('❌ Missing item code');
                alert('Error: Item code is missing');
                return;
            }
            
            console.log('📦 Item data:', {itemCode, itemName, uom, warehouse});
            
            let poItemData = null;
            try {
                poItemData = JSON.parse(poItemJson);
            } catch (e) {
                console.warn('Could not parse PO item data:', e);
            }
            
            // Call the function
            addItemToGRPO(itemCode, itemName, uom, warehouse, poItemData);
        });
    });
    
    console.log('✅ Event listeners set up complete');
});
